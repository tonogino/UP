UP_CHI_update_mao_mechanic_gui = {
	clamp_variable = {
		var = UP_CHI_mao_support
		min = 0
		max = 100
	}
	clamp_variable = {
		var = UP_CHI_mao_suspicion
		min = 0
		max = 100
	}

	set_variable = { UP_CHI_mao_support_pointer_x = UP_CHI_mao_support }
	multiply_variable = { UP_CHI_mao_support_pointer_x = 3.18 }
	add_to_variable = { UP_CHI_mao_support_pointer_x = 8 }

	set_variable = { UP_CHI_mao_suspicion_pointer_x = UP_CHI_mao_suspicion }
	multiply_variable = { UP_CHI_mao_suspicion_pointer_x = 3.18 }
	add_to_variable = { UP_CHI_mao_suspicion_pointer_x = 8 }

	add_to_variable = { UP_CHI_mao_mechanic_dirty = 1 }

	if = {
		limit = {
			check_variable = { UP_CHI_mao_suspicion > 59 }
			has_country_flag = UP_CHI_mao_crisis_enabled
			NOT = { has_country_flag = UP_CHI_mao_suspicion_60_triggered }
		}
		set_country_flag = UP_CHI_mao_suspicion_60_triggered
		country_event = { id = up_asia.340 days = 1 }
	}
	if = {
		limit = {
			check_variable = { UP_CHI_mao_suspicion > 79 }
			has_country_flag = UP_CHI_mao_crisis_enabled
			NOT = { has_country_flag = UP_CHI_mao_suspicion_80_triggered }
		}
		set_country_flag = UP_CHI_mao_suspicion_80_triggered
		country_event = { id = up_asia.341 days = 1 }
	}
	if = {
		limit = {
			check_variable = { UP_CHI_mao_suspicion > 89 }
			has_country_flag = UP_CHI_mao_crisis_enabled
			NOT = { has_country_flag = UP_CHI_mao_suspicion_90_triggered }
		}
		set_country_flag = UP_CHI_mao_suspicion_90_triggered
		country_event = { id = up_asia.342 days = 1 }
	}
	if = {
		limit = {
			check_variable = { UP_CHI_mao_suspicion > 99 }
			has_country_flag = UP_CHI_mao_crisis_enabled
			NOT = { has_country_flag = UP_CHI_cultural_revolution_started }
		}
		set_country_flag = UP_CHI_cultural_revolution_started
		country_event = { id = up_asia.350 days = 1 }
	}
}

UP_CHI_mao_daily_balance_effect = {
	if = {
		limit = {
			has_country_flag = UP_CHI_mao_mechanic_active
			has_country_flag = UP_CHI_mao_crisis_enabled
			NOT = { has_country_flag = UP_CHI_assassination_resolved }
			NOT = { has_country_flag = UP_CHI_cultural_revolution_started }
		}
		if = {
			limit = { check_variable = { UP_CHI_mao_support > 69 } }
			add_to_variable = { UP_CHI_mao_suspicion = -0.03 }
		}
		if = {
			limit = {
				check_variable = { UP_CHI_mao_support > 49 }
				check_variable = { UP_CHI_mao_support < 70 }
			}
			add_to_variable = { UP_CHI_mao_suspicion = -0.01 }
		}
		if = {
			limit = {
				check_variable = { UP_CHI_mao_support > 29 }
				check_variable = { UP_CHI_mao_support < 50 }
			}
			add_to_variable = { UP_CHI_mao_suspicion = 0.01 }
		}
		if = {
			limit = { check_variable = { UP_CHI_mao_support < 30 } }
			add_to_variable = { UP_CHI_mao_suspicion = 0.03 }
		}
	}

	if = {
		limit = {
			has_country_flag = UP_CHI_mao_crisis_enabled
			NOT = { has_country_flag = UP_CHI_assassination_resolved }
			NOT = { has_country_flag = UP_CHI_cultural_revolution_started }
			check_variable = { UP_CHI_mao_support > 69 }
		}
		if = { limit = { NOT = { has_idea = UP_mao_support_high } } add_ideas = UP_mao_support_high }
		remove_ideas = UP_mao_support_above_average
		remove_ideas = UP_mao_support_below_average
		remove_ideas = UP_mao_support_low
	}
	if = {
		limit = {
			has_country_flag = UP_CHI_mao_crisis_enabled
			NOT = { has_country_flag = UP_CHI_assassination_resolved }
			NOT = { has_country_flag = UP_CHI_cultural_revolution_started }
			check_variable = { UP_CHI_mao_support > 49 }
			check_variable = { UP_CHI_mao_support < 70 }
		}
		if = { limit = { NOT = { has_idea = UP_mao_support_above_average } } add_ideas = UP_mao_support_above_average }
		remove_ideas = UP_mao_support_high
		remove_ideas = UP_mao_support_below_average
		remove_ideas = UP_mao_support_low
	}
	if = {
		limit = {
			has_country_flag = UP_CHI_mao_crisis_enabled
			NOT = { has_country_flag = UP_CHI_assassination_resolved }
			NOT = { has_country_flag = UP_CHI_cultural_revolution_started }
			check_variable = { UP_CHI_mao_support > 29 }
			check_variable = { UP_CHI_mao_support < 50 }
		}
		if = { limit = { NOT = { has_idea = UP_mao_support_below_average } } add_ideas = UP_mao_support_below_average }
		remove_ideas = UP_mao_support_high
		remove_ideas = UP_mao_support_above_average
		remove_ideas = UP_mao_support_low
	}
	if = {
		limit = {
			has_country_flag = UP_CHI_mao_crisis_enabled
			NOT = { has_country_flag = UP_CHI_assassination_resolved }
			NOT = { has_country_flag = UP_CHI_cultural_revolution_started }
			check_variable = { UP_CHI_mao_support < 30 }
		}
		if = { limit = { NOT = { has_idea = UP_mao_support_low } } add_ideas = UP_mao_support_low }
		remove_ideas = UP_mao_support_high
		remove_ideas = UP_mao_support_above_average
		remove_ideas = UP_mao_support_below_average
	}

	UP_CHI_update_mao_mechanic_gui = yes
}

UP_CHI_cultural_revolution_monthly_effect = {
	add_manpower = -75000
	add_stability = -0.005
	add_to_variable = { UP_CHI_cultural_revolution_months = 1 }
	if = {
		limit = {
			check_variable = { UP_CHI_cultural_revolution_months > 2 }
		}
		set_variable = { UP_CHI_cultural_revolution_months = 0 }
		if = {
			limit = {
				NOT = { has_country_flag = UP_CHI_cultural_revolution_event_1_fired }
				NOT = { has_country_flag = UP_CHI_cultural_revolution_event_chain_locked }
			}
			set_country_flag = UP_CHI_cultural_revolution_event_chain_locked
			set_country_flag = UP_CHI_cultural_revolution_event_1_fired
			country_event = { id = up_asia.360 days = 1 }
		}
		if = {
			limit = {
				has_country_flag = UP_CHI_cultural_revolution_event_1_fired
				NOT = { has_country_flag = UP_CHI_cultural_revolution_event_2_fired }
				NOT = { has_country_flag = UP_CHI_cultural_revolution_event_chain_locked }
			}
			set_country_flag = UP_CHI_cultural_revolution_event_chain_locked
			set_country_flag = UP_CHI_cultural_revolution_event_2_fired
			country_event = { id = up_asia.361 days = 1 }
		}
		if = {
			limit = {
				has_country_flag = UP_CHI_cultural_revolution_event_2_fired
				NOT = { has_country_flag = UP_CHI_cultural_revolution_event_3_fired }
				NOT = { has_country_flag = UP_CHI_cultural_revolution_event_chain_locked }
			}
			set_country_flag = UP_CHI_cultural_revolution_event_chain_locked
			set_country_flag = UP_CHI_cultural_revolution_event_3_fired
			country_event = { id = up_asia.362 days = 1 }
		}
		if = {
			limit = {
				has_country_flag = UP_CHI_cultural_revolution_event_3_fired
				NOT = { has_country_flag = UP_CHI_cultural_revolution_event_4_fired }
				NOT = { has_country_flag = UP_CHI_cultural_revolution_event_chain_locked }
			}
			set_country_flag = UP_CHI_cultural_revolution_event_chain_locked
			set_country_flag = UP_CHI_cultural_revolution_event_4_fired
			country_event = { id = up_asia.363 days = 1 }
		}
		if = {
			limit = {
				has_country_flag = UP_CHI_cultural_revolution_event_4_fired
				NOT = { has_country_flag = UP_CHI_cultural_revolution_event_5_fired }
				NOT = { has_country_flag = UP_CHI_cultural_revolution_event_chain_locked }
			}
			set_country_flag = UP_CHI_cultural_revolution_event_chain_locked
			set_country_flag = UP_CHI_cultural_revolution_event_5_fired
			country_event = { id = up_asia.364 days = 1 }
		}
	}
	clr_country_flag = UP_CHI_cultural_revolution_event_chain_locked
}

UP_CHI_end_cultural_revolution_standard = {
	if = {
		limit = {
			OR = {
				has_country_flag = UP_CHI_cultural_revolution_started
				has_idea = UP_cultural_revolution_chaos
			}
		}
		clr_country_flag = UP_CHI_cultural_revolution_started
		clr_country_flag = UP_CHI_mao_crisis_enabled
		remove_ideas = UP_cultural_revolution_chaos
		remove_ideas = UP_cr_red_guard_turbulence
		remove_ideas = UP_cr_industrial_disruption
		remove_ideas = UP_cr_army_politicization
		if = {
			limit = {
				NOT = { has_idea = UP_cultural_revolution_aftermath }
				NOT = { has_idea = UP_cultural_revolution_revolutionary_legacy }
			}
			add_ideas = UP_cultural_revolution_aftermath
		}
		set_country_flag = UP_CHI_cultural_revolution_resolved
		country_event = { id = up_asia.365 days = 1 }
	}
}

UP_CHI_start_mao_mechanic = {
	set_country_flag = UP_CHI_mao_mechanic_active
	if = {
		limit = { NOT = { has_country_flag = UP_CHI_mao_mechanic_initialized } }
		set_country_flag = UP_CHI_mao_mechanic_initialized
		set_variable = { UP_CHI_mao_support = 50 }
		set_variable = { UP_CHI_mao_suspicion = 10 }
	}
	set_variable = { UP_CHI_mao_mechanic_dirty = 1 }
	UP_CHI_update_mao_mechanic_gui = yes
	UP_CHI_mao_daily_balance_effect = yes
}
