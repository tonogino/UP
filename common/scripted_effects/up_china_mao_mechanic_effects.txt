UP_CHI_update_mao_mechanic_gui = {
	clamp_variable = {
		var = UP_CHI_mao_support
		min = 0
		max = 100
	}
	clamp_variable = {
		var = UP_CHI_mao_suspicion
		min = 0
		max = 100
	}

	set_variable = { UP_CHI_mao_support_pointer_x = UP_CHI_mao_support }
	multiply_variable = { UP_CHI_mao_support_pointer_x = 3.18 }
	add_to_variable = { UP_CHI_mao_support_pointer_x = 8 }

	set_variable = { UP_CHI_mao_suspicion_pointer_x = UP_CHI_mao_suspicion }
	multiply_variable = { UP_CHI_mao_suspicion_pointer_x = 3.18 }
	add_to_variable = { UP_CHI_mao_suspicion_pointer_x = 8 }

	add_to_variable = { UP_CHI_mao_mechanic_dirty = 1 }

	if = {
		limit = {
			check_variable = { UP_CHI_mao_suspicion > 59 }
			has_country_flag = UP_CHI_mao_crisis_enabled
			NOT = { has_country_flag = UP_CHI_mao_suspicion_60_triggered }
		}
		set_country_flag = UP_CHI_mao_suspicion_60_triggered
		country_event = { id = up_asia.340 days = 1 }
	}
	if = {
		limit = {
			check_variable = { UP_CHI_mao_suspicion > 79 }
			has_country_flag = UP_CHI_mao_crisis_enabled
			NOT = { has_country_flag = UP_CHI_mao_suspicion_80_triggered }
		}
		set_country_flag = UP_CHI_mao_suspicion_80_triggered
		country_event = { id = up_asia.341 days = 1 }
	}
	if = {
		limit = {
			check_variable = { UP_CHI_mao_suspicion > 89 }
			has_country_flag = UP_CHI_mao_crisis_enabled
			NOT = { has_country_flag = UP_CHI_mao_suspicion_90_triggered }
		}
		set_country_flag = UP_CHI_mao_suspicion_90_triggered
		country_event = { id = up_asia.342 days = 1 }
	}
	if = {
		limit = {
			check_variable = { UP_CHI_mao_suspicion > 99 }
			has_country_flag = UP_CHI_mao_crisis_enabled
			NOT = { has_country_flag = UP_CHI_cultural_revolution_started }
		}
		set_country_flag = UP_CHI_cultural_revolution_started
		country_event = { id = up_asia.350 days = 1 }
	}
}

UP_CHI_cultural_revolution_monthly_effect = {
	add_manpower = -75000
	add_stability = -0.005
	add_to_variable = { UP_CHI_cultural_revolution_months = 1 }
	if = {
		limit = {
			check_variable = { UP_CHI_cultural_revolution_months > 2 }
			check_variable = { UP_CHI_cultural_revolution_events_fired < 5 }
		}
		set_variable = { UP_CHI_cultural_revolution_months = 0 }
		add_to_variable = { UP_CHI_cultural_revolution_events_fired = 1 }
		if = { limit = { check_variable = { UP_CHI_cultural_revolution_events_fired = 1 } } country_event = { id = up_asia.360 days = 1 } }
		if = { limit = { check_variable = { UP_CHI_cultural_revolution_events_fired = 2 } } country_event = { id = up_asia.361 days = 1 } }
		if = { limit = { check_variable = { UP_CHI_cultural_revolution_events_fired = 3 } } country_event = { id = up_asia.362 days = 1 } }
		if = { limit = { check_variable = { UP_CHI_cultural_revolution_events_fired = 4 } } country_event = { id = up_asia.363 days = 1 } }
		if = { limit = { check_variable = { UP_CHI_cultural_revolution_events_fired = 5 } } country_event = { id = up_asia.364 days = 1 } }
	}
}

UP_CHI_start_mao_mechanic = {
	set_country_flag = UP_CHI_mao_mechanic_active
	if = {
		limit = { NOT = { has_country_flag = UP_CHI_mao_mechanic_initialized } }
		set_country_flag = UP_CHI_mao_mechanic_initialized
		set_variable = { UP_CHI_mao_support = 50 }
		set_variable = { UP_CHI_mao_suspicion = 10 }
	}
	set_variable = { UP_CHI_mao_mechanic_dirty = 1 }
	UP_CHI_update_mao_mechanic_gui = yes
}
