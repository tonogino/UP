UP_JAP_clear_imperial_conference_dominance = {
	clr_country_flag = UP_JAP_dominant_kodoha
	clr_country_flag = UP_JAP_dominant_toseiha
	clr_country_flag = UP_JAP_dominant_navy
	clr_country_flag = UP_JAP_dominant_empire
}

UP_JAP_update_imperial_conference_gui = {
	clamp_variable = {
		var = UP_JAP_kodoha_prestige
		min = 0
		max = 999
	}
	clamp_variable = {
		var = UP_JAP_toseiha_prestige
		min = 0
		max = 999
	}
	clamp_variable = {
		var = UP_JAP_navy_prestige
		min = 0
		max = 999
	}
	clamp_variable = {
		var = UP_JAP_fiscal_pressure
		min = 0
		max = 100
	}

	UP_JAP_update_fiscal_pressure_spirit = yes

	UP_JAP_clear_imperial_conference_dominance = yes

	if = {
		limit = {
			check_variable = { UP_JAP_kodoha_prestige > 69 }
			check_variable = { UP_JAP_toseiha_prestige > 69 }
			check_variable = { UP_JAP_navy_prestige > 69 }
		}
		set_country_flag = UP_JAP_dominant_empire
	}
	else_if = {
		limit = {
			check_variable = { UP_JAP_kodoha_prestige > UP_JAP_toseiha_prestige }
			check_variable = { UP_JAP_kodoha_prestige > UP_JAP_navy_prestige }
		}
		set_country_flag = UP_JAP_dominant_kodoha
	}
	else_if = {
		limit = {
			check_variable = { UP_JAP_toseiha_prestige > UP_JAP_kodoha_prestige }
			check_variable = { UP_JAP_toseiha_prestige > UP_JAP_navy_prestige }
		}
		set_country_flag = UP_JAP_dominant_toseiha
	}
	else_if = {
		limit = {
			check_variable = { UP_JAP_navy_prestige > UP_JAP_kodoha_prestige }
			check_variable = { UP_JAP_navy_prestige > UP_JAP_toseiha_prestige }
		}
		set_country_flag = UP_JAP_dominant_navy
	}

	add_to_variable = { UP_JAP_imperial_conference_dirty = 1 }
}

UP_JAP_start_imperial_conference = {
	set_country_flag = UP_JAP_imperial_conference_active
	if = {
		limit = { NOT = { has_country_flag = UP_JAP_imperial_conference_initialized } }
		set_country_flag = UP_JAP_imperial_conference_initialized
		set_variable = { UP_JAP_kodoha_prestige = 45 }
		set_variable = { UP_JAP_toseiha_prestige = 35 }
		set_variable = { UP_JAP_navy_prestige = 30 }
		set_variable = { UP_JAP_fiscal_pressure = 55 }
		add_ideas = {
			UP_JAP_imperial_overload
			UP_JAP_far_eastern_security_war
			UP_JAP_zaibatsu_creditors
		}
	}
	UP_JAP_update_imperial_conference_gui = yes
}

UP_JAP_update_fiscal_pressure_spirit = {
	remove_ideas = {
		UP_JAP_fiscal_plenty
		UP_JAP_fiscal_controlled_deficit
		UP_JAP_fiscal_tension
		UP_JAP_fiscal_crisis
		UP_JAP_fiscal_collapse
	}
	if = {
		limit = { check_variable = { UP_JAP_fiscal_pressure < 21 } }
		add_ideas = UP_JAP_fiscal_plenty
	}
	else_if = {
		limit = { check_variable = { UP_JAP_fiscal_pressure < 41 } }
		add_ideas = UP_JAP_fiscal_controlled_deficit
	}
	else_if = {
		limit = { check_variable = { UP_JAP_fiscal_pressure < 61 } }
		add_ideas = UP_JAP_fiscal_tension
	}
	else_if = {
		limit = { check_variable = { UP_JAP_fiscal_pressure < 81 } }
		add_ideas = UP_JAP_fiscal_crisis
	}
	else = {
		add_ideas = UP_JAP_fiscal_collapse
	}
}
