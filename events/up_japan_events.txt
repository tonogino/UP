		UP_JAP_update_imperial_conference_gui = yes
	}
	option = {
		name = up_japan.6.b
		UP_JAP_reduce_fiscal_pressure_large = yes
		UP_JAP_reduce_navy_prestige_large = yes
		add_timed_idea = {
			idea = UP_JAP_naval_budget_frozen
			days = 365
		}
	}
	option = {
		name = up_japan.6.c
		UP_JAP_reduce_fiscal_pressure_large = yes
		UP_JAP_reduce_toseiha_prestige_large = yes
		XIC = {
			add_ideas = UP_unleashed_puppet
		}
	}
	option = {
		name = up_japan.6.e
		UP_JAP_reduce_fiscal_pressure_massive = yes
		add_ideas = UP_JAP_zaibatsu_controlled_economy
		add_stability = -0.10
	}
	option = {
		name = up_japan.6.f
		add_stability = -0.15
		if = {
			limit = { has_country_flag = UP_JAP_dominant_kodoha }
			set_country_flag = UP_JAP_route_kodoha_crisis
		}
		else_if = {
			limit = { has_country_flag = UP_JAP_dominant_toseiha }
			set_country_flag = UP_JAP_route_toseiha_crisis
		}
		else_if = {
			limit = { has_country_flag = UP_JAP_dominant_navy }
			set_country_flag = UP_JAP_route_navy_crisis
		}
		else = {
			set_country_flag = UP_JAP_showa_integration_failed
		}
	}
}
